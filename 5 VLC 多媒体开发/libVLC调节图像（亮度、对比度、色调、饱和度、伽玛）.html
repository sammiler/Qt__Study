本文介绍了如何使用libVLC库调整视频的亮度、对比度、色调、饱和度和伽玛，以改善画面效果。文章详细阐述了这些参数的作用，并提供了核心API及示例程序来演示参数的设置方法。

对于一个视频来说，色彩和画面效果的呈现非常重要。假如你的画面偏暗或偏亮，缺乏层次感，色彩不够丰富或不自然，则需要根据场景和氛围进行调整。

所涉及的重要参数有：

  • 亮度： 是视频画面的明暗程度。调整亮度可以使画面看起来更舒适、更真实。（需要调整亮度的情况：画面过暗或过亮）

  • 对比度： 是指同一画面中黑色和白色之间的差异对比度，即画面的明暗对比度。适当的对比度可以使画面更清晰、更有层次感。过高或过低的对比度会影响画面的质量和观感效果。（需要调整对比度的情况：画面缺乏层次感或过于对比鲜明）

  • 色调： 是指视频画面中的颜色属于冷色或暖色等不同色调系列。调整色调可以改变画面的整体色彩氛围，进而影响观众的情感体验。（需要调整色调的情况：画面整体色调过暗或过亮、需要根据情境做出不同的色调）

  • 饱和度： 是指视频画面中的颜色饱和程度。适当的饱和度可以提高画面的色彩鲜艳度，但过高的饱和度会使画面变得过于艳丽，不自然，影响观感效果。（需要调整饱和度的情况：包括画面颜色过于单调或等颜色不够饱和。）

  • 伽玛： 是指视频的灰度响应曲线，即视频信号的输出亮度与输入亮度之间的关系。伽马调整可以改变画面的整体亮度和对比度。（需要调整伽马的情况：包括画面的局部明亮区域或暗区域需要调整，或画面整体色彩不协调需要调整）

值得注意的是，不同的调整参数对画面效果的影响是相互关联的，正确的调整参数需要根据具体情况进行综合考虑。

核心 API

有了以上概念的了解，来看看 libVLC 中对各个参数的定义：

enum libvlc_video_adjust_option_t {
            
    libvlc_adjust_Enable = 0,
    libvlc_adjust_Contrast,  // 对比度
    libvlc_adjust_Brightness,  // 亮度
    libvlc_adjust_Hue,  // 色调
    libvlc_adjust_Saturation,  // 饱和度
    libvlc_adjust_Gamma  // 伽玛
};

根据《VLC Documentation:Modules/adjust》描述，各参数的取值范围如下：

  • 对比度：Contrast (0.0 ~ 2.0)，默认值：1.0
  • 亮度：Brightness (0.0 ~ 2.0)，默认值：1.0
  • 色调：Hue (-180 ~ 180)，默认值：0
  • 饱和度：Saturation (0.0 ~ 3.0)，默认值：1.0
  • 伽玛：Gamma (0.01 ~ 10.0)，默认值：1.0

要设置它们，可以通过以下接口来完成：

/**
 * 设置要调整的选项
 *
 * 参数 p_mi：媒体播放器实例
 * 参数 option：待调整的选项，取值见 libvlc_video_adjust_option_t
 * 参数 value：待调整选项的值
 */
LIBVLC_API void libvlc_video_set_adjust_float( libvlc_media_player_t *p_mi,
                                                   unsigned option, float value );

示例程序

下面，来播放一个本地视频，并通过 spin box 来调整具体的值：

#pragma once

#include <QWidget>

class QLabel;
class QDoubleSpinBox;
class libvlc_instance_t;
class libvlc_media_player_t;

class VideoAdjust : public QWidget
{
            
    Q_OBJECT

public:
    VideoAdjust(QWidget *parent = Q_NULLPTR);
    ~VideoAdjust();
    void play(const QString &file);

private slots:
    void adjustContrast(double value);
    void adjustBrightness(double value);
    void adjustHue(double value);
    void adjustSaturation(double value);
    void adjustGamma(double value);

private:
    void playMedia(const QString &file);
    void initData();

private:
    QLabel *label;
    QLabel *contrastLabel;
    QLabel *brightnessLabel;
    QLabel *hueLabel;
    QLabel *saturationLabel;
    QLabel *gammaLabel;
    QDoubleSpinBox *contrastSpinBox;
    QDoubleSpinBox *brightnessSpinBox;
    QDoubleSpinBox *hueSpinBox;
    QDoubleSpinBox *saturationSpinBox;
    QDoubleSpinBox *gammaSpinBox;

    libvlc_instance_t *vlcInstance;
    libvlc_media_player_t *mediaPlayer;
};

具体的参数调节是由 adjustXXX() 槽函数实现的：

#include "VideoAdjust.h"
#include <QLabel>
#include <QDoubleSpinBox>
#include <QVBoxLayout>

// 避免 ssize_t 引发的错误
#if defined(_MSC_VER)
#include <BaseTsd.h>
typedef SSIZE_T ssize_t;
#endif

#include <vlc/vlc.h>

VideoAdjust::VideoAdjust(QWidget *parent)
    : QWidget(parent)
{
            
    resize(400, 300);

    label = new QLabel(this);
    contrastLabel = new QLabel(this);
    brightnessLabel = new QLabel(this);
    hueLabel = new QLabel(this);
    saturationLabel = new QLabel(this);
    gammaLabel = new QLabel(this);
    contrastSpinBox = new QDoubleSpinBox(this);
    brightnessSpinBox = new QDoubleSpinBox(this);
    hueSpinBox = new QDoubleSpinBox(this);
    saturationSpinBox = new QDoubleSpinBox(this);
    gammaSpinBox = new QDoubleSpinBox(this);

    contrastLabel->setText(QStringLiteral("对比度"));
    brightnessLabel->setText(QStringLiteral("亮度"));
    hueLabel->setText(QStringLiteral("色调"));
    saturationLabel->setText(QStringLiteral("饱和度"));
    gammaLabel->setText(QStringLiteral("伽玛"));

    contrastSpinBox->setRange(0.0, 2.0);
    brightnessSpinBox->setRange(0.0, 2.0);
    hueSpinBox->setRange(-180, 180);
    saturationSpinBox->setRange(0.0, 3.0);
    gammaSpinBox->setRange(0.01, 10.0);

    contrastSpinBox->setSingleStep(0.1);
    brightnessSpinBox->setSingleStep(0.1);
    hueSpinBox->setSingleStep(1);
    saturationSpinBox->setSingleStep(0.1);
    gammaSpinBox->setSingleStep(0.01);

    QGridLayout *gridLayout = new QGridLayout();
    gridLayout->addWidget(contrastLabel, 0, 0);
    gridLayout->addWidget(contrastSpinBox, 0, 1);
    gridLayout->addWidget(brightnessLabel, 1, 0);
    gridLayout->addWidget(brightnessSpinBox, 1, 1);
    gridLayout->addWidget(hueLabel, 2, 0);
    gridLayout->addWidget(hueSpinBox, 2, 1);
    gridLayout->addWidget(saturationLabel, 3, 0);
    gridLayout->addWidget(saturationSpinBox, 3, 1);
    gridLayout->addWidget(gammaLabel, 4, 0);
    gridLayout->addWidget(gammaSpinBox, 4, 1);
    gridLayout->setHorizontalSpacing(10);
    gridLayout->setVerticalSpacing(10);
    gridLayout->setContentsMargins(0, 0, 0, 0);

    QVBoxLayout *layout = new QVBoxLayout();
    layout->addWidget(label);
    layout->addLayout(gridLayout);
    layout->setSpacing(10);
    layout->setContentsMargins(10, 10, 10, 10);
    setLayout(layout);

    connect(contrastSpinBox, SIGNAL(valueChanged(double)), this, SLOT(adjustContrast(double)));
    connect(brightnessSpinBox, SIGNAL(valueChanged(double)), this, SLOT(adjustBrightness(double)));
    connect(hueSpinBox, SIGNAL(valueChanged(double)), this, SLOT(adjustHue(double)));
    connect(saturationSpinBox, SIGNAL(valueChanged(double)), this, SLOT(adjustSaturation(double)));
    connect(gammaSpinBox, SIGNAL(valueChanged(double)), this, SLOT(adjustGamma(double)));
}

VideoAdjust::~VideoAdjust()
{
            
    // 停止播放
    libvlc_media_player_stop(mediaPlayer);

    // 释放 media player
    libvlc_media_player_release(mediaPlayer);

    // 释放 libvlc 实例
    libvlc_release(vlcInstance);
}

void VideoAdjust::play(const QString &file)
{
            
    playMedia(file);
    initData();
}

void VideoAdjust::playMedia(const QString &file)
{
            
    libvlc_media_t *media;

    // 创建并初始化 libvlc 实例
    vlcInstance = libvlc_new(0, NULL);

    // 创建一个 media，参数是一个媒体资源位置（例如：有效的 URL）。
    media = libvlc_media_new_path(vlcInstance, file.toUtf8().data());

    // 创建一个 media player 播放环境
    mediaPlayer = libvlc_media_player_new_from_media(media);

    // 设置输出窗口（将视频渲染到 Qt 窗口中）
    libvlc_media_player_set_hwnd(mediaPlayer, (void*)(label->winId()));

    // 现在，不需要保留 media 了
    libvlc_media_release(media);

    // 播放 media player
    libvlc_media_player_play(mediaPlayer);

    // 让图像可调节
    libvlc_video_set_adjust_int(mediaPlayer, libvlc_adjust_Enable, 1);
}

void VideoAdjust::initData()
{
            
    float contrast = libvlc_video_get_adjust_float(mediaPlayer, libvlc_adjust_Contrast);
    float brightness = libvlc_video_get_adjust_float(mediaPlayer, libvlc_adjust_Brightness);
    int hue = libvlc_video_get_adjust_int(mediaPlayer, libvlc_adjust_Hue);
    float saturation = libvlc_video_get_adjust_float(mediaPlayer, libvlc_adjust_Saturation);
    float gamma = libvlc_video_get_adjust_float(mediaPlayer, libvlc_adjust_Gamma);

    contrastSpinBox->setValue(contrast);
    brightnessSpinBox->setValue(brightness);
    hueSpinBox->setValue(hue);
    saturationSpinBox->setValue(saturation);
    gammaSpinBox->setValue(gamma);
}

void VideoAdjust::adjustContrast(double value)
{
            
    libvlc_video_set_adjust_float(mediaPlayer, libvlc_adjust_Contrast, value);
}

void VideoAdjust::adjustBrightness(double value)
{
            
    libvlc_video_set_adjust_float(mediaPlayer, libvlc_adjust_Brightness, value);
}

void VideoAdjust::adjustHue(double value)
{
            
    libvlc_video_set_adjust_int(mediaPlayer, libvlc_adjust_Hue, value);
}

void VideoAdjust::adjustSaturation(double value)
{
            
    libvlc_video_set_adjust_float(mediaPlayer, libvlc_adjust_Saturation, value);
}

void VideoAdjust::adjustGamma(double value)
{
            
    libvlc_video_set_adjust_float(mediaPlayer, libvlc_adjust_Gamma, value);
}
