本文介绍了如何使用 libVLC 在播放视频时添加图片和文本水印，通过设置 logo 和文本数据结构及接口，实现水印在视频上的位置调整和透明度控制。提供了一个简单的示例程序展示如何在左上角添加图片水印，在右上角添加文本水印。

在使用 VLC 播放视频时，如果想添加 logo 或水印，可以使用“效果及滤镜”。这使用的是叠加视频效果，此效果将允许我们加载 JPG、PNG 等图片。

这里的设置比较丰富，不仅能自定义 logo 在视频上的位置，还可以设置该水印的透明度，以便它不会挡住当前播放的视频文件。如果愿意，还可以添加文本叠加层。

当然，使用 libVLC 实现这样的功能也很简单，先来看一下效果图（左上角显示图片水印，右上角显示文本水印）：

数据结构

在 libVLC 中，logo 和文本的控制主要涉及以下数据结构和接口。

logo

logo 选项由 libvlc_video_logo_option_t 表示， 主要用于 libvlc_video_{get,set}_logo_{int,string} 的接口调用，以获取和设置 logo。

/** 用于 libvlc_video_{get,set}_logo_{int,string} 的选项值*/
enum libvlc_video_logo_option_t {
            
    libvlc_logo_enable,         // 是否启用 logo
    libvlc_logo_file,           // logo 对应的文件名
    libvlc_logo_x,              // logo 的 x 坐标
    libvlc_logo_y,              // logo 的 y 坐标
    libvlc_logo_delay,          // logo 显示间隔时间 0 - 60000 毫秒。
    libvlc_logo_repeat,         // 标识 logo 的循环（动画的循环数量，-1 表示继续，0 表示关闭）
    libvlc_logo_opacity,        // logo 的透明度（0 - 255，0 表示完全透明，255 表示完全不透明）
    libvlc_logo_position        // 1（左）、2（右）、4（頂部）、8 （底部），也可以混合使用，例如：6=4+2 表示右上。
};

文本

而文本选项由 libvlc_video_marquee_option_t 表示， 主要用于 libvlc_video_{get,set}_marquee_{int,string} 的接口调用，以获取和设置文本。

typedef enum libvlc_video_marquee_option_t {
            
    libvlc_marquee_Enable = 0,  // 是否启用文本
    libvlc_marquee_Text,        // 文本
    libvlc_marquee_Color,       // 文本颜色
    libvlc_marquee_Opacity,     // 文本透明度
    libvlc_marquee_Position,    // 文本位置
    libvlc_marquee_Refresh,     // 文本刷新
    libvlc_marquee_Size,        // 文本字体大小
    libvlc_marquee_Timeout,     // 文本超时时间（ms）
    libvlc_marquee_X,           // 文本的 X 坐标
    libvlc_marquee_Y            // 文本的 Y 坐标
} libvlc_video_marquee_option_t;

示例程序

在了解了这些核心数据结构和接口之后，我们就可以着手写示例程序了，最简单的方式就是将《最简单的 libVLC 播放器》中的代码改造一下。

首先，我们添加一个显示图片水印的接口，用于将 logo 显示在视频的左上角。

void showLogo(libvlc_media_player_t *mp, const char * logo)
{
            
    libvlc_video_set_logo_int(mp, libvlc_logo_enable, 1);  // 启用 logo
    libvlc_video_set_logo_string(mp, libvlc_logo_file, logo);  // logo 的地址
    // libvlc_video_set_logo_int(mp, libvlc_logo_position, 5);  // 显示在左上方
    libvlc_video_set_logo_int(mp, libvlc_logo_x, 10);  // x 坐标
    libvlc_video_set_logo_int(mp, libvlc_logo_y, 10);  // y 坐标
    libvlc_video_set_logo_int(mp, libvlc_logo_delay, 2000);  // 间隔时间
    libvlc_video_set_logo_int(mp, libvlc_logo_repeat, -1);  // 一直显示
    libvlc_video_set_logo_int(mp, libvlc_logo_opacity, 200);  // 透明度
}

紧接着，再添加一个显示文本水印的接口，用于将文本显示在视频的右上角。

void showMarquee(libvlc_media_player_t *mp, const char *text)
{
            
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Enable, 1);  // 启用文本
    libvlc_video_set_marquee_string(mp, libvlc_marquee_Text, text);  // 文本
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Position, 6); // 显示在右上方
    libvlc_video_set_marquee_int(mp, libvlc_marquee_X, 10);  // x 坐标
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Y, 10);  // y 坐标
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Opacity, 200);  // 透明度
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Refresh, 1);
    // libvlc_video_set_marquee_int(mp, libvlc_marquee_Timeout, 10000);  // 文本超时时间（ms）
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Size, 64);  // 文本大小
    libvlc_video_set_marquee_int(mp, libvlc_marquee_Color, 0xEE9572);  // 文本颜色
}

最后，在播放视频时调用这两个接口：

int main()
{
            
    const char * localMrl = "F:\\Videos\\Sample.mkv";
    const char * logo = "F:\\Videos\\logo.jpg";
    const char * text = "Waleon";

    libvlc_instance_t *instance;
    libvlc_media_player_t *player;
    libvlc_media_t *media;

    // 创建并初始化 libvlc 实例
    instance = libvlc_new(0, nullptr);

    // 创建一个媒体及播放器
    media = libvlc_media_new_path(instance, localMrl);
    player = libvlc_media_player_new_from_media(media);

    // 播放 media player
    libvlc_media_player_play(player);

    // 显示 logo
    showLogo(player, logo);

    // 显示文本
    showMarquee(player, text);

    // 播放媒体
    waitPlaying(player);

    // 释放资源
    libvlc_media_release(media);
    libvlc_media_player_release(player);
    libvlc_release(instance);

    return 0;
}

这样就完成了图片和文本水印的显示了，效果如上图所示，很简单，有木有。
